001package edu.pdx.cs410J.hui2;
002
003import edu.pdx.cs410J.AbstractPhoneBill;
004import edu.pdx.cs410J.AbstractPhoneCall;
005import edu.pdx.cs410J.PhoneBillDumper;
006
007import java.io.BufferedWriter;
008import java.io.FileWriter;
009import java.io.IOException;
010
011/**
012 * The class that will implement the fucntions of writing the command line arguments to the text file
013 */
014public class TextDumper implements PhoneBillDumper {
015    String fileName;
016    String customerName;
017    String callerNumber;
018    String calleeNumber;
019    String startTime;
020    String endTime;
021    AbstractPhoneCall newPhoneCall;
022    AbstractPhoneBill totalPhoneBill;
023
024    /**
025     * This function is to write the information form the command line into the text file. It will also
026     * check if the object is null, if it is then we want to create a new object of phonebill and store
027     * the information.
028     * @param bill the phonebill that pass in from the command line
029     * @throws IOException it will throw an exception if the file location doesn't exit to create the file name
030     */
031    @Override
032    public void dump(AbstractPhoneBill bill) throws IOException
033    {
034        AbstractPhoneBill phonebill = null;
035        BufferedWriter bw = null;
036        FileWriter fw = null;
037        bw = new BufferedWriter(new FileWriter(fileName, true));
038        bw.write(customerName + "," + callerNumber + "," + calleeNumber + "," + startTime + "," + endTime);
039        bw.newLine();
040        bw.flush();
041        if(bw!=null)
042            bw.close();
043        if(bill == null) {
044            phonebill = new PhoneBill(customerName, new PhoneCall(callerNumber, calleeNumber, startTime, endTime));
045            bill = phonebill;
046        }
047        else {
048            newPhoneCall = new PhoneCall(callerNumber, calleeNumber, startTime, endTime);
049            bill.addPhoneCall(newPhoneCall);
050        }
051        totalPhoneBill = bill;
052
053    }
054
055    /**
056     * gets the file name
057     * @param fileName store the file name
058     */
059    public void getFile(String fileName)
060    {
061        this.fileName = fileName;
062    }
063
064    /**
065     * a constructor to take in the variables
066     * @param customer string contains customer name
067     * @param caller string contains the caller's phone number
068     * @param callee string contains the callee's phone number
069     * @param startTime string contains the start date and time
070     * @param endTime string contains the end date and time
071     */
072    TextDumper(String customer, String caller, String callee, String startTime, String endTime)
073    {
074        this.customerName = customer;
075        this.callerNumber = caller;
076        this.calleeNumber = callee;
077        this.startTime = startTime;
078        this.endTime = endTime;
079    }
080
081    /**
082     *
083     * @return it returns the phonebill
084     */
085    public AbstractPhoneBill returnPhoneBill()
086    {
087        return totalPhoneBill;
088    }
089}
